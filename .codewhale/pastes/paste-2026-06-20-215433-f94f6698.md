## 4️⃣ `server.js`（核心改造：输出 JSON 布局）

```javascript
require('dotenv').config();
const express = require('express');
const fetch = require('node-fetch');
const path = require('path');

const app = express();
app.use(express.json());
app.use(express.static(path.join(__dirname, 'public')));

const PORT = process.env.PORT || 3000;
const ENABLE_AI = process.env.ENABLE_AI === 'true';
const API_KEY = process.env.API_KEY;
const API_URL = process.env.API_URL;
const API_MODEL = process.env.API_MODEL;

if (ENABLE_AI) {
    if (!API_KEY || !API_URL || !API_MODEL) {
        console.error('❌ ENABLE_AI=true 但缺少 API_KEY / API_URL / API_MODEL');
        process.exit(1);
    }
    console.log('✅ AI 功能已启用，模型:', API_MODEL);
} else {
    console.log('ℹ️  AI 功能未启用，将使用内置模拟 UI 布局');
}

// 生成 UI 布局接口（支持历史对话）
app.post('/api/generate-ui', async (req, res) => {
    const { instruction, history } = req.body;
    if (!instruction) {
        return res.status(400).json({ error: '缺少指令' });
    }

    try {
        let layout;
        if (ENABLE_AI) {
            const messages = [
                {
                    role: 'system',
                    content: `你是一个 UI 生成器。根据用户指令生成一个 XP 风格的窗口布局 JSON。
                    必须返回一个 JSON 对象，格式如下：
                    {
                      "title": "窗口标题",
                      "components": [
                        // 组件对象数组
                      ]
                    }
                    每个组件对象包含 "type" 字段，支持以下类型：
                    - "text": { "type": "text", "content": "字符串" }
                    - "button": { "type": "button", "label": "字符串", "action": "提示文本或JS代码" }
                    - "input": { "type": "input", "placeholder": "字符串", "id": "唯一ID" }
                    - "image": { "type": "image", "src": "图片URL", "alt": "替代文本" }
                    - "list": { "type": "list", "items": ["项目1", "项目2", ...] }
                    - "grid": { "type": "grid", "rows": [[单元格内容,...], ...] }
                    除了类型字段，每个组件可以有 "style" 对象（如 {"color": "blue"}）和 "events" 对象（如 {"onClick": "alert('hello')"}），但 events 仅限简单 JS 代码。
                    只返回 JSON，不要任何解释。`
                },
                ...(history || []),   // 支持多轮对话
                { role: 'user', content: instruction }
            ];

            const response = await fetch(API_URL, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${API_KEY}`
                },
                body: JSON.stringify({
                    model: API_MODEL,
                    messages,
                    temperature: 0.7,
                    max_tokens: 2000
                })
            });

            if (!response.ok) throw new Error(`AI API 错误: ${response.status}`);
            const data = await response.json();
            const text = data.choices[0].message.content.trim();

            // 清理可能的 markdown 代码块
            let jsonStr = text;
            if (jsonStr.startsWith('```')) {
                jsonStr = jsonStr.replace(/```json?|```/g, '').trim();
            }
            layout = JSON.parse(jsonStr);
        } else {
            // 模拟 UI 布局
            layout = getMockLayout(instruction);
        }
        res.json(layout);
    } catch (error) {
        console.error('UI 生成失败:', error.message);
        if (ENABLE_AI) {
            // 出错时也返回模拟布局，保证演示不中断
            res.json(getMockLayout(instruction));
        } else {
            res.status(500).json({ error: '生成失败' });
        }
    }
});

// 模拟布局函数
function getMockLayout(instruction) {
    const lower = instruction.toLowerCase();
    if (lower.includes('天气') || lower.includes('weather')) {
        return {
            title: '天气面板',
            components: [
                { type: 'text', content: '☀️ 今天晴天，25°C', style: { fontSize: '18px', fontWeight: 'bold' } },
                { type: 'button', label: '刷新', action: 'alert("刷新天气")' },
                { type: 'list', items: ['周一: 晴 26°', '周二: 多云 24°', '周三: 雨 21°'] }
            ]
        };
    } else if (lower.includes('待办') || lower.includes('todo')) {
        return {
            title: '待办清单',
            components: [
                { type: 'input', placeholder: '添加新任务...', id: 'todoInput' },
                { type: 'button', label: '添加', action: 'addTodo()' },
                { type: 'list', items: ['完成报告', '整理文件', '回复邮件'] }
            ]
        };
    } else {
        return {
            title: '示例组件',
            components: [
                { type: 'text', content: '这是一个动态生成的 XP 风格窗口。试试输入“显示天气”或“待办清单”。' },
                { type: 'button', label: '点我', action: 'alert("你好！")' },
                { type: 'image', src: 'https://via.placeholder.com/150', alt: '占位图' }
            ]
        };
    }
}

// 首页
app.get('*', (req, res) => {
    res.sendFile(path.join(__dirname, 'public', 'index.html'));
});

app.listen(PORT, () => {
    console.log(`🖥️  Windows XP AI Desktop (Dynamic UI) 运行在 http://localhost:${PORT}`);
});
```

    <script>
        // ========== 基础 UI ==========
        const desktop = document.getElementById('desktop');
        const startBtn = document.getElementById('start-btn');
        const startMenu = document.getElementById('start-menu');
        const clockEl = document.getElementById('clock');
        const container = document.getElementById('windows-container');

        let startVisible = false;
        let windows = [];
        let activeId = null;
        let zCounter = 500;

        // 时钟
        function updateClock() {
            const d = new Date();
            const h = d.getHours(), m = d.getMinutes();
            clockEl.textContent = `${h%12||12}:${String(m).padStart(2,'0')} ${h>=12?'下午':'上午'}`;
        }
        updateClock(); setInterval(updateClock, 30000);

        // 开始菜单
        startBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            startVisible = !startVisible;
            startMenu.classList.toggle('visible', startVisible);
        });
        document.addEventListener('click', () => { startVisible = false; startMenu.classList.remove('visible'); });
        startMenu.querySelector('.menu-item').addEventListener('click', () => {
            startMenu.classList.remove('visible');
            createDynamicWindow('未命名窗口', []);
        });

        // 创建动态窗口
        function createDynamicWindow(title, components, history = []) {
            const id = 'win-' + Date.now();
            const z = ++zCounter;
            const win = { id, title, components, history, element: null, x: 60 + windows.length*30, y: 50 + windows.length*30 };

            const el = document.createElement('div');
            el.className = 'ai-window visible active';
            el.id = id;
            el.style.left = win.x + 'px';
            el.style.top = win.y + 'px';
            el.style.zIndex = z;
            el.innerHTML = `
                <div class="win-titlebar">
                    <span>${escapeHtml(title || '新窗口')}</span>
                    <button class="win-close">✕</button>
                </div>
                <div class="win-body" id="body-${id}"></div>
                <div class="win-input-bar">
                    <input type="text" placeholder="输入指令修改此窗口..." id="input-${id}">
                    <button id="send-${id}">发送</button>
                </div>
            `;
            container.appendChild(el);
            win.element = el;

            // 渲染组件
            renderComponents(id, components);

            // 事件
            el.querySelector('.win-close').addEventListener('click', () => {
                el.remove(); windows = windows.filter(w => w.id !== id);
            });
            el.addEventListener('mousedown', () => activateWindow(id));

            // 指令发送
            const input = el.querySelector(`#input-${id}`);
            const sendBtn = el.querySelector(`#send-${id}`);
            sendBtn.addEventListener('click', () => handleInstruction(id));
            input.addEventListener('keydown', (e) => { if (e.key === 'Enter') handleInstruction(id); });

            // 拖拽
            const titlebar = el.querySelector('.win-titlebar');
            titlebar.addEventListener('mousedown', (e) => {
                if (e.target.tagName === 'BUTTON') return;
                const startX = e.clientX, startY = e.clientY;
                const origLeft = el.offsetLeft, origTop = el.offsetTop;
                const move = (ev) => {
                    el.style.left = (origLeft + ev.clientX - startX) + 'px';
                    el.style.top = (origTop + ev.clientY - startY) + 'px';
                };
                const up = () => {
                    document.removeEventListener('mousemove', move);
                    document.removeEventListener('mouseup', up);
                };
                document.addEventListener('mousemove', move);
                document.addEventListener('mouseup', up);
            });

            windows.push(win);
            activateWindow(id);
            return id;
        }

        function activateWindow(id) {
            windows.forEach(w => w.element.classList.remove('active'));
            const win = windows.find(w => w.id === id);
            if (win) {
                win.element.classList.add('active');
                win.element.style.zIndex = ++zCounter;
                activeId = id;
            }
        }

        // 渲染组件
        function renderComponents(winId, components) {
            const body = document.getElementById(`body-${winId}`);
            if (!body) return;
            body.innerHTML = '';
            components.forEach(comp => {
                const el = createComponentElement(comp);
                if (el) body.appendChild(el);
            });
        }

        function createComponentElement(comp) {
            const type = comp.type;
            const style = comp.style || {};
            const events = comp.events || {};
            let el;

            switch (type) {
                case 'text':
                    el = document.createElement('div');
                    el.className = 'dyn-text';
                    el.textContent = comp.content || '';
                    break;
                case 'button':
                    el = document.createElement('button');
                    el.className = 'dyn-btn';
                    el.textContent = comp.label || '按钮';
                    if (comp.action) {
                        el.addEventListener('click', () => {
                            try { eval(comp.action); } catch(e) { alert(comp.action); }
                        });
                    }
                    break;
                case 'input':
                    el = document.createElement('input');
                    el.type = 'text';
                    el.className = 'dyn-input';
                    el.placeholder = comp.placeholder || '';
                    el.id = comp.id || '';
                    break;
                case 'list':
                    el = document.createElement('ul');
                    el.className = 'dyn-list';
                    (comp.items || []).forEach(item => {
                        const li = document.createElement('li');
                        li.textContent = item;
                        el.appendChild(li);
                    });
                    break;
                case 'grid':
                    el = document.createElement('table');
                    el.className = 'dyn-grid';
                    (comp.rows || []).forEach(row => {
                        const tr = document.createElement('tr');
                        row.forEach(cell => {
                            const td = document.createElement('td');
                            td.textContent = cell;
                            tr.appendChild(td);
                        });
                        el.appendChild(tr);
                    });
                    break;
                case 'image':
                    el = document.createElement('img');
                    el.src = comp.src || '';
                    el.alt = comp.alt || '';
                    el.style.maxWidth = '100%';
                    break;
                default:
                    el = document.createElement('div');
                    el.textContent = '未知组件';
            }

            // 应用样式
            Object.entries(style).forEach(([key, val]) => {
                el.style[key] = val;
            });
            return el;
        }

        // 处理指令
        async function handleInstruction(winId) {
            const input = document.getElementById(`input-${winId}`);
            if (!input) return;
            const instruction = input.value.trim();
            if (!instruction) return;
            input.value = '';
            const win = windows.find(w => w.id === winId);
            if (!win) return;

            // 状态提示
            const body = document.getElementById(`body-${winId}`);
            body.innerHTML = '<div style="color:#888;">⏳ AI 正在生成界面...</div>';

            try {
                const res = await fetch('/api/generate-ui', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        instruction,
                        history: win.history || []
                    })
                });
                const layout = await res.json();
                // 更新窗口
                win.title = layout.title || instruction;
                win.components = layout.components || [];
                win.element.querySelector('.win-titlebar span').textContent = win.title;
                // 更新历史
                win.history = [
                    ...(win.history || []),
                    { role: 'user', content: instruction },
                    { role: 'assistant', content: JSON.stringify(layout) }
                ];
                renderComponents(winId, win.components);
            } catch (err) {
                body.innerHTML = '<div style="color:red;">生成失败，请重试</div>';
                console.error(err);
            }
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }

        // 初始背景回退
        const bgImg = new Image();
        bgImg.onerror = () => desktop.classList.add('fallback-bg');
        bgImg.src = 'https://upload.wikimedia.org/wikipedia/commons/0/05/Bliss_%28Windows_XP%29.png';
        setTimeout(() => { if (!bgImg.complete || bgImg.naturalWidth===0) desktop.classList.add('fallback-bg'); }, 4000);
    </script>
</body>

## 🚀 运行方式

1. 将以上文件覆盖到项目目录（注意备份原文件）。
2. 安装依赖：`npm install`
3. 启动：`npm start`
4. 访问 `http://localhost:3000`

### 如何体验动态 UI？
- 点击 **开始** → **✨ 新建动态窗口**。
- 在窗口底部输入指令，例如：
  - “显示一个天气面板”
  - “做一个带按钮的待办清单”
  - “生成一张图片和一个输入框”
- 如果启用了真实 AI，模型会返回 JSON 布局并渲染；否则使用内置模拟布局。

### 多轮对话修改
- 在同一个窗口继续输入新指令，AI 会结合之前生成的历史进行调整（例如“把按钮改成蓝色”）。

## 🎯 核心改变对照

| 旧版（文本浏览器） | 新版（动态 UI 生成器） |
|-------------------|------------------------|
| AI 返回 Markdown 文章 | AI 返回 JSON 组件树 |
| IE 窗口渲染静态内容 | XP 窗口实时渲染可交互组件 |
| 搜索 → 文章 | 指令 → 界面 |
| 无对话历史 | 窗口内支持连续修改 |
